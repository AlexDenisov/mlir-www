28 /// Return the SSA values that represent the data point accessed using a given
29 /// `indexingMap` for a given point in the iteration space represented by `ivs`.
30 static SmallVector<Value> getIndicesForAccess(OpBuilder &b, Location loc,
31  AffineMap indexingMap,
32  ValueRange ivs) {
33  SmallVector<Value> indices;
34  indices.reserve(indexingMap.getNumResults());
35  for (auto result : indexingMap.getResults()) {
36  AffineMap m = AffineMap::get(indexingMap.getNumDims(),
37  indexingMap.getNumSymbols(), result);
38  Value v = b.create<AffineApplyOp>(loc, m, ivs);
39  indices.push_back(v);
40  }
41  return indices;
42 }

44 /// Method to inline the payload of a `linalgOp` given the iteration space
45 /// point and values for the arguments of the payload.
46 static LogicalResult inlinePayload(OpBuilder &b, LinalgOp linalgOp,
47  ValueRange ivs, ValueRange argValues) {
48  Block *body = linalgOp.getBlock();
49  BlockAndValueMapping map;
50  map.map(body->getArguments(), argValues);
51  for (auto &op : body->without_terminator()) {
52  if (auto indexOp = dyn_cast<IndexOp>(&op)) {
53  map.map(indexOp.getResult(), ivs[indexOp.getDim()]);
54  continue;
55  }
56  b.clone(op, map);
57  }
58 
59  Operation *terminator = body->getTerminator();
60  Location loc = terminator->getLoc();
61  for (const auto &operand : llvm::enumerate(terminator->getOperands())) {
62  Value toStore = map.lookupOrDefault(operand.value());
63  OpOperand *storeInto = linalgOp.getDpsInitOperand(operand.index());
64  auto indices = getIndicesForAccess(
65  b, loc, linalgOp.getMatchingIndexingMap(storeInto), ivs);
66  b.create<memref::StoreOp>(
67  loc, toStore, linalgOp.getDpsInitOperand(operand.index())->get(),
68  indices);
69  }
70  return success();
71 }

78 /// External model implementation of TilingInterface for LinalgOps. An external
79 /// model implementation is used for now till the use of `TilingInterface` is
80 /// on-par with the current Linalg tiling + fusion patterns. Once it is
81 /// maybe possible to move this into the op-definition (though there are
82 /// advantages to leaving it as an external model)
83 template <typename LinalgOpTy>
84 struct LinalgOpTilingInterface
85  : public TilingInterface::ExternalModel<LinalgOpTilingInterface<LinalgOpTy>,
86  LinalgOpTy> {
87  /// Return the loop iterator type.
88  SmallVector<utils::IteratorType> getLoopIteratorTypes(Operation *op) const {
89  LinalgOpTy concreteOp = cast<LinalgOpTy>(op);
90  return llvm::to_vector(llvm::map_range(
91  concreteOp.getIteratorTypesArray(), [](StringRef iteratorType) {
92  return utils::symbolizeIteratorType(iteratorType).value();
93  }));

96  /// Return the iteration domain range.
97  SmallVector<Range> getIterationDomain(Operation *op, OpBuilder &b) const {
98  OpBuilder::InsertionGuard g(b);
99  b.setInsertionPoint(op);
100  Location loc = op->getLoc();
101  LinalgOp linalgOp = cast<LinalgOp>(op);
102  SmallVector<OpFoldResult> allShapesSizes =
103  linalgOp.createFlatListOfOperandDims(b, loc);
104  AffineMap map = linalgOp.getShapesToLoopsMap();
105 
106  return llvm::to_vector(
107  llvm::map_range(map.getResults(), [&](AffineExpr loopExpr) {
108  OpFoldResult ofr =
109  makeComposedFoldedAffineApply(b, loc, loopExpr, allShapesSizes);
110  return Range{b.getIndexAttr(0), ofr, b.getIndexAttr(1)};
111  }));
112  }

114  // Instantiate the tiled implementation of the operation.
115  SmallVector<Operation *>
116  getTiledImplementation(Operation *op, OpBuilder &b,
117  ArrayRef<OpFoldResult> offsets,
118  ArrayRef<OpFoldResult> sizes) const {
119  // Leave the `sizeBounds` value empty. That is only needed when the `sizes`
120  // specified could lead to out of bounds accesses.
121  Location loc = op->getLoc();
122  LinalgOp linalgOp = cast<LinalgOp>(op);
123  SmallVector<Value> valuesToTile = linalgOp->getOperands();
124  SmallVector<Value, 4> tiledOperands = makeTiledShapes(
125  b, loc, linalgOp, valuesToTile, offsets, sizes, {}, true);
126 
127  SmallVector<Type> resultTensorTypes =
128  getTensorOutputTypes(linalgOp, tiledOperands);
129 
130  Operation *tiledOp =
131  linalgOp.clone(b, loc, resultTensorTypes, tiledOperands);
132  offsetIndices(b, cast<LinalgOp>(tiledOp), offsets);
133 
134  return {tiledOp};
135  }

137  // Return the details of the output tile generated by the tiled
138  // implementation.
139  LogicalResult
140  getResultTilePosition(Operation *op, OpBuilder &b, unsigned resultNumber,
141  ArrayRef<OpFoldResult> offsets,
142  ArrayRef<OpFoldResult> sizes,
143  SmallVector<OpFoldResult> &resultOffsets,
144  SmallVector<OpFoldResult> &resultSizes) const {
145  Location loc = op->getLoc();
146  LinalgOp linalgOp = cast<LinalgOp>(op);
147 
148  AffineExpr d0;
149  bindDims(b.getContext(), d0);
150  SmallVector<OpFoldResult> subShapeSizes =
151  llvm::to_vector(llvm::map_range(sizes, [&](OpFoldResult ofr) {
152  return makeComposedFoldedAffineApply(b, loc, d0 - 1, ofr);
153  }));
154 
155  OpOperand *outOperand = linalgOp.getDpsInitOperand(resultNumber);
156  SliceParameters sliceParams = computeSliceParameters(
157  b, loc, outOperand->get(), sizes,
158  linalgOp.getMatchingIndexingMap(outOperand), offsets,
159  /*ubs*/ {}, subShapeSizes, true);
160  resultOffsets = sliceParams.offsets;
161  resultSizes = sliceParams.sizes;
162  return success();
163  }

165  FailureOr<Value> generateResultTileValue(Operation *op, OpBuilder &b,
166  unsigned resultNumber,
167  ArrayRef<OpFoldResult> offsets,
168  ArrayRef<OpFoldResult> sizes) const {
169  auto linalgOp = cast<LinalgOp>(op);
170 
171  // Check that the indexing map used for the output is a projected
172  // permutation. This could be relaxed with a more general approach that can
173  // map the offsets and sizes from the result to iteration space tiles
174  // (filling in full extent for dimensions not used to access the result).
175  AffineMap indexingMap =
176  linalgOp.getIndexingMapMatchingResult(op->getResult(resultNumber));
177  if (!indexingMap.isProjectedPermutation()) {
178  return op->emitOpError(
179  "unhandled tiled implementation generation when result is not "
180  "accessed using a permuted projection");
181  }
182 
183  auto numLoops = linalgOp.getNumLoops();
184  auto tilingInterfaceOp = cast<TilingInterface>(op);
185  SmallVector<OpFoldResult> iterationTileOffsets(numLoops),
186  iterationTileSizes(numLoops);
187  if (!indexingMap.isPermutation()) {
188  SmallVector<Range> iterationDomain =
189  tilingInterfaceOp.getIterationDomain(b);
190  for (const auto &range : llvm::enumerate(iterationDomain)) {
191  iterationTileOffsets[range.index()] = range.value().offset;
192  iterationTileSizes[range.index()] = range.value().size;
193  }
194  }
195  for (const auto &resultExpr : llvm::enumerate(indexingMap.getResults())) {
196  unsigned dimPosition =
197  resultExpr.value().cast<AffineDimExpr>().getPosition();
198  iterationTileOffsets[dimPosition] = offsets[resultExpr.index()];
199  iterationTileSizes[dimPosition] = sizes[resultExpr.index()];
200  }
201 
202  SmallVector<Operation *> tiledOp = tilingInterfaceOp.getTiledImplementation(
203  b, iterationTileOffsets, iterationTileSizes);
204  if (tiledOp.size() != 1)
205  return op->emitOpError("failed to generate tiled implementation");
206 
207  return tiledOp[0]->getResult(resultNumber);
208  }

210  LogicalResult generateScalarImplementation(Operation *op, OpBuilder &builder,
211  Location loc,
212  ValueRange ivs) const {
213  auto linalgOp = cast<LinalgOp>(op);
214  if (!linalgOp.hasBufferSemantics())
215  return op->emitOpError("expected operation to have buffer semantics");
216 
217  SmallVector<Value> indexedValues;
218  indexedValues.reserve(linalgOp->getNumOperands());
219  Location linalgOpLoc = op->getLoc();
220  /// Load the data corresponding to the block arguments that
221  /// represent input operands.
222  for (OpOperand &operand : linalgOp->getOpOperands()) {
223  if (!linalgOp.payloadUsesValueFromOperand(&operand)) {
224  indexedValues.push_back(nullptr);
225  continue;
226  }
227  if (linalgOp.isScalar(&operand)) {
228  indexedValues.push_back(operand.get());
229  continue;
230  }
231  SmallVector<Value> indices = getIndicesForAccess(
232  builder, linalgOpLoc, linalgOp.getMatchingIndexingMap(&operand), ivs);
233  Value load =
234  builder.create<memref::LoadOp>(linalgOpLoc, operand.get(), indices);
235  indexedValues.push_back(load);
236  }
237 
238  /// Inline the op payload and store the result.
239  return inlinePayload(builder, linalgOp, ivs, indexedValues);
240  }
